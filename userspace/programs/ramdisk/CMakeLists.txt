generate_nanopb_proto(
    PROTO_SRCS
    PROTO_HEADERS
    ${CMAKE_SOURCE_DIR}/proto/mosrpc.proto
    ${CMAKE_SOURCE_DIR}/proto/blockdev.proto
)

    PRIVATE
        blockdev-manager-lib
        nanopb
        librpc::client
        librpc::server
)
